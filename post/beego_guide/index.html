Beego使用指南

2022年7月31日
| Beego
| 本文阅读量 次
文章目录

这篇文章是 Beego 框架的使用教程，实现了如何用 bee 工具自动生成项目的基本结构。

Beego的安装

要想构建 Beego 项目需要下载安装 Beego 框架和 Bee 工具。

Beego

Beego 框架的安装很简单，只需要执行下面的代码就行:

1go get -u github.com/beego/beego/v2

这样 Beego 框架就安装好了。

Bee

想要快速创建项目，还需要安装 Bee 工具:

1go get -u github.com/beego/bee/v2

安装好了 Bee 工具之后，需要将 Bee 所在目录 $GOPATH/bin 加入到 $PATH 环境变量中，Linux 系统可以直接将二进制文件 bee 复制到 /usr/local/bin/ 目录下。

快速入门

Beego 是一个能够进行快速开发应用的框架。

model

首先登陆mysql，创建数据库my_test。

1mysql> CREATE DATABASE my_test

在models文件夹下新建model.go，写入代码：

 1package models
 2
 3import (
 4	"github.com/beego/beego/v2/client/orm"
 5	"github.com/beego/beego/v2/core/logs"
 6	_ "github.com/go-sql-driver/mysql"
 7)
 8
 9// User 用户信息
10type User struct {
11	Id   int
12	Name string `orm:"unique"` // 用户名唯一
13	Pwd  string
14}
15
16func init() {
17	// 设置数据库基本信息
18	if err := orm.RegisterDataBase("default", "mysql", "root:password@tcp(127.0.0.1:3306)/my_test?charset=utf8"); err != nil {
19		logs.Error(err)
20	}
21	// 映射model数据库
22	orm.RegisterModel(&User{})
23	// 生成表
24	if err := orm.RunSyncdb("default", false, true); err != nil {
25		logs.Error(err)
26	}
27}

controller

在 controllers 文件夹下的 default.go 中写入代码：

 1package controllers
 2
 3import (
 4	"beego-demo/models"
 5	"github.com/beego/beego/v2/client/orm"
 6	"github.com/beego/beego/v2/core/logs"
 7	beego "github.com/beego/beego/v2/server/web"
 8	"net/http"
 9	"strings"
10)
11
12type MainController struct {
13	beego.Controller
14}
15
16// RegisterGet 注册展示页面
17func (c *MainController) RegisterGet() {
18	c.TplName = "register.html"
19}
20
21// RegisterPost 注册获取数据页面
22func (c *MainController) RegisterPost() {
23	// 1.拿到数据，并去除两边的空格
24	username := strings.TrimSpace(c.GetString("username"))
25	password := strings.TrimSpace(c.GetString("password"))
26	// 2.对数据进行校验
27	if username == "" || password == "" {
28		logs.Error("用户名或密码不能为空")
29		c.Redirect("/register", http.StatusFound)
30		return
31	}
32	// 3.插入数据库
33	o := orm.NewOrm()
34	user := models.User{
35		Name: username,
36		Pwd:  password,
37	}
38	_, err := o.Insert(&user)
39	if err != nil {
40		logs.Error("insert failed, err:", err)
41		c.Redirect("/register", http.StatusNotFound)
42		return
43	}
44	// 4.返回登陆界面
45	// c.TplName = "login.html" // 指定视图文件，同时可以给这个视图传递一些数据
46	c.Redirect("/login", http.StatusFound) // 跳转，不能传递数据，速度快
47}

routers

最后在 routers 文件夹的 router.go 文件中写入如下代码：

 1package routers
 2
 3import (
 4	"beego-demo/controllers"
 5	beego "github.com/beego/beego/v2/server/web"
 6)
 7
 8func init() {
 9	beego.Router("/register", &controllers.MainController{}, "get:RegisterGet;post:RegisterPost")
10}

这样，注册功能就简单实现了。

model

model.go 文件没有新增内容。

controller

在 controllers 文件夹下的 default.go 中写入代码：

 1package controllers
 2
 3import (
 4	"beego-demo/models"
 5	"github.com/beego/beego/v2/client/orm"
 6	"github.com/beego/beego/v2/core/logs"
 7	beego "github.com/beego/beego/v2/server/web"
 8	"net/http"
 9	"strings"
10)
11
12type MainController struct {
13	beego.Controller
14}
15
16// RegisterGet 注册展示页面
17func (c *MainController) RegisterGet() {
18	c.TplName = "register.html"
19}
20
21// RegisterPost 注册获取数据页面
22func (c *MainController) RegisterPost() {
23	// 1.拿到数据，并去除两边的空格
24	username := strings.TrimSpace(c.GetString("username"))
25	password := strings.TrimSpace(c.GetString("password"))
26	// 2.对数据进行校验
27	if username == "" || password == "" {
28		logs.Error("用户名或密码不能为空")
29		c.Redirect("/register", http.StatusFound)
30		return
31	}
32	// 3.插入数据库
33	o := orm.NewOrm()
34	user := models.User{
35		Name: username,
36		Pwd:  password,
37	}
38	_, err := o.Insert(&user)
39	if err != nil {
40		logs.Error("insert failed, err:", err)
41		c.Redirect("/register", http.StatusNotFound)
42		return
43	}
44	// 4.返回登陆界面
45	// c.TplName = "login.html" // 指定视图文件，同时可以给这个视图传递一些数据
46	c.Redirect("/login", http.StatusFound) // 跳转，不能传递数据，速度快
47}
48
49// LoginGet 登陆页面get方法
50func (c *MainController) LoginGet() {
51	c.TplName = "login.html"
52}
53
54// LoginPost 登陆页面post方法
55func (c *MainController) LoginPost() {
56	// 1.拿到数据，并去除两边的空格
57	username := strings.TrimSpace(c.GetString("username"))
58	password := strings.TrimSpace(c.GetString("password"))
59	logs.Info("username:", username, "password:", password)
60	// 2.判断数据是否合法
61	if username == "" || password == "" {
62		logs.Error("用户名或密码不能为空")
63		c.TplName = "login.html"
64		c.Data["errMsg"] = "登陆失败"
65	}
66	// 3.查询账号密码是否正确
67	o := orm.NewOrm()
68	user := models.User{}
69	user.Name = username
70	if err := o.Read(&user, "Name"); err != nil {
71		logs.Info("用户名错误")
72		c.TplName = "login.html"
73		c.Data["errMsg"] = "用户名或密码错误"
74		return
75	}
76	// 判断密码是否一致
77	if user.Pwd != password {
78		logs.Info("密码错误")
79		c.TplName = "login.html"
80		c.Data["errMsg"] = "用户名或密码错误"
81		return
82	}
83	// 4.跳转
84	c.Ctx.WriteString("登陆成功")
85}

routers

最后在 routers 文件夹的 router.go 文件中写入如下代码：

 1package routers
 2
 3import (
 4	"beego-demo/controllers"
 5	beego "github.com/beego/beego/v2/server/web"
 6)
 7
 8func init() {
 9	beego.Router("/register", &controllers.MainController{}, "get:RegisterGet;post:RegisterPost")
10    beego.Router("/login", &controllers.MainController{}, "get:LoginGet;post:LoginPost")
11}

就这样，登陆功能也实现了。

效果展示

在浏览器中打开 localhost:8000/register 可进入注册界面。

输入用户名和密码后，点击注册，注册成功会跳转到登录界面。在登陆界面输入正确的用户名和密码后，就会跳转到登陆成功界面。

这样，利用 Beego 框架就实现了简单的登陆和注册功能。
